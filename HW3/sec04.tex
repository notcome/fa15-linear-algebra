\begin{exercise}{9}
  \begin{enumerate}
    \item \begin{align*}
            (f \circ g \circ h)(a) &= f(g(h(a))) = f(g(h(3))) \\
                                   &= f(g(-3+2)) = f(g(-1)) \\
                                   &= f(3 \times (-1)) = f(-3) \\
                                   &= (-3)^2-1 = 8
          \end{align*}
    \item \begin{align*}
            (f \circ g \circ h)(a) &= f(g(h(a))) = f(g(h(1))) \\
                                   &= f(g(1-3)) = f(g(-2)) \\
                                   &= f(-2-3) = f(-5) \\
                                   &= (-5)^2 = 25
          \end{align*}
  \end{enumerate}
\end{exercise}
